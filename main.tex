\chapter{شرکا و همکاران پیشنهادی}
برای اجرای موفق این کلان‌طرح، لازم است ترکیبی از نهادهای سیاست‌گذار، دستگاه‌های اجرایی، تشکل‌های تخصصی، زنجیره‌های تولید، مراکز علمی و بازیگران بازار به‌صورت هماهنگ مشارکت داشته باشند.

\section{نهاد اصلی راهبری طرح}
\begin{itemize}
    \item \textbf{وزارت جهاد کشاورزی:} به‌عنوان متولی اصلی امنیت غذایی، تنظیم تولید، تأمین نهاده، سیاست‌گذاری جوجه‌ریزی و مدیریت زنجیره گوشت مرغ.
    \item \textbf{معاونت امور تولیدات دامی وزارت جهاد کشاورزی:} برای هدایت تخصصی بخش طیور.
    \item \textbf{شرکت پشتیبانی امور دام کشور:} برای تأمین، ذخیره‌سازی، مداخله هوشمند در بازار و مدیریت ذخایر راهبردی.
\end{itemize}

\section{نهادهای تنظیم‌گر و پشتیبان دولتی}
\begin{itemize}
    \item \textbf{سازمان برنامه و بودجه کشور} برای تأمین، تخصیص و نظارت بر بودجه ۳۰ میلیارد تومانی طرح.
    \item \textbf{وزارت صنعت، معدن و تجارت} برای هماهنگی در شبکه توزیع، پایش بازار و جلوگیری از اختلال در عرضه.
    \item \textbf{سازمان حمایت مصرف‌کنندگان و تولیدکنندگان} برای پایش قیمت، جلوگیری از گران‌فروشی و حمایت هم‌زمان از تولیدکننده و مصرف‌کننده.
    \item \textbf{بانک مرکزی جمهوری اسلامی ایران} برای تسهیل تأمین ارز، مدیریت ابزارهای مالی و رفع گلوگاه‌های پرداخت در واردات نهاده.
    \item \textbf{گمرک جمهوری اسلامی ایران} برای تسریع ترخیص نهاده‌های دامی، کاهش رسوب کالا و جلوگیری از تأخیر در تأمین.
\end{itemize}

\section{نهادهای نظارتی و حاکمیتی}

    \item \textbf{شورای عالی امنیت غذایی} برای پیوند دادن طرح با سیاست‌های کلان امنیت غذایی کشور.
    \item \textbf{سازمان تعزیرات حکومتی} برای برخورد با احتکار، عرضه خارج از شبکه و تخلفات مؤثر بر قیمت.
    \item \textbf{سازمان بازرسی کل کشور} برای نظارت بر سلامت فرایند تخصیص نهاده، قراردادها و اجرای طرح.

\section{تشکل‌ها و اتحادیه‌های تخصصی بخش طیور}

    \item \textbf{اتحادیه سراسری مرغ تخم‌گذار و مادر} در بخش تأمین جوجه یک‌روزه و پایداری چرخه تولید.
    \item \textbf{انجمن جوجه یک‌روزه کشور} برای هماهنگی ظرفیت جوجه‌ریزی و پیش‌بینی عرضه آینده.
    \item \textbf{انجمن صنایع خوراک دام، طیور و آبزیان ایران} برای ارتقای کیفیت و ثبات تأمین خوراک.

    \item \textbf{میادین میوه و تره‌بار، فروشگاه‌های زنجیره‌ای و سامانه‌های فروش آنلاین} برای توزیع شفاف و قابل پایش گوشت مرغ.

\chapter{بررسی سابقه اجرا و مطالعه این طرح در ایران}
بر اساس شواهد موجود از اسناد شما و همچنین نتایج جست‌وجوی تکمیلی، می‌توان گفت که \textbf{ایران سابقه اجرای پراکنده اما متعددِ سیاست‌ها و مطالعات مرتبط با تنظیم بازار گوشت مرغ، تأمین نهاده، و مدیریت زنجیره تولید طیور} را داشته است؛ اما این اقدامات غالباً \textbf{جزیره‌ای، مقطعی و بدون یکپارچگی نهادی کامل} بوده‌اند.
بنابراین، طرح ملی پیشنهادی شما در امتداد تجربه‌های قبلی کشور قرار می‌گیرد، اما تفاوت آن در \textbf{یکپارچه‌سازی حکمرانی، تثبیت تولید، و اتصال نهاده تا بازار} است.

\section{سابقه سیاستی: تمرکز بر نهاده به‌جای مدیریت زنجیره}
در سال‌های گذشته، بخش مهمی از مداخلات دولت برای تنظیم بازار مرغ، از مسیر \textbf{تأمین و توزیع نهاده‌های دامی} انجام شده است. سامانه \textbf{بازارگاه} به‌عنوان ابزار اصلی تخصیص نهاده، با هدف شفاف‌سازی و جلوگیری از انحراف توزیع راه‌اندازی شد. این تجربه نشان می‌دهد که در ایران، نگاه سیاستی عمدتاً بر \textbf{کنترل ورودی‌های تولید} متمرکز بوده، نه مدیریت یکپارچه کل زنجیره از جوجه‌ریزی تا توزیع نهایی.

\section{سابقه اجرایی: مداخله‌های مقطعی برای کنترل بازار}
در مقاطع مختلف، دولت و نهادهای تنظیم‌گر برای مهار نوسان قیمت مرغ، اقداماتی مانند:
\begin{itemize}
    \item عرضه نهاده با نرخ ترجیحی،
    \item توزیع مدت‌دار،
    \item مداخله شرکت پشتیبانی امور دام،
    \item و کنترل جوجه یک‌روزه در بازارگاه
\end{itemize}
را به‌کار گرفته‌اند. این تجربه‌ها نشان می‌دهد که \textbf{ابزار تنظیم بازار مرغ در ایران وجود داشته، اما به‌دلیل نبود هماهنگی کامل میان حلقه‌های زنجیره، ثبات پایدار ایجاد نکرده است.}

\section{سابقه مطالعاتی: توجه به زنجیره یکپارچه و بهره‌وری}
در مطالعات و گزارش‌های مختلف، بر \textbf{زنجیره‌های یکپارچه طیور} به‌عنوان الگوی کارآمدتر تولید تأکید شده است. این مطالعات معمولاً نشان می‌دهند که زنجیره‌های یکپارچه:
\begin{itemize}
    \item توان بیشتری در مدیریت نهاده دارند،
    \item در بحران‌ها تاب‌آورترند،
    \item هزینه تمام‌شده پایین‌تری دارند،
    \item و می‌توانند در ثبات عرضه نقش مهم‌تری ایفا کنند.
\end{itemize}
با این حال، این ظرفیت‌ها در سیاست‌گذاری عمومی کشور به‌طور کامل بالفعل نشده‌اند.

\section{تجربه‌های تنظیم بازار جوجه یک‌روزه و نهاده}
در سال‌های اخیر، سیاست‌هایی برای \textbf{عرضه جوجه یک‌روزه در بازارگاه} و کنترل بهتر مرحله جوجه‌ریزی نیز اجرا یا تصویب شده است. این موضوع نشان می‌دهد که درک سیاستی به‌تدریج از «تنظیم صرفِ نهاده» به سمت «تنظیم مرحله‌ای زنجیره تولید» حرکت کرده است. با این وجود، نبود سامانه یکپارچه داده و وزن‌دهی متناسب به واحدهای بزرگ و کوچک، مانع تحقق کامل این هدف شده است.

\section{جمع‌بندی از سابقه ایران}
بنابراین، سابقه ایران در این حوزه را می‌توان چنین خلاصه کرد:
\begin{itemize}
    \item \textbf{مطالعه و تجربه وجود دارد؛}
    \item \textbf{ابزارهای اجرایی نیز وجود داشته‌اند؛}
    \item اما \textbf{طرح جامع، منسجم، و پایدار ملی برای ثبات زنجیره گوشت مرغ} هنوز به‌صورت کامل و یکپارچه اجرا نشده است.
\end{itemize}
